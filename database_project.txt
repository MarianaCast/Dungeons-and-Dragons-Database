CREATE TABLE `armor` (
  `Armor_Class` varchar(45) NOT NULL,
  `Type` varchar(45) DEFAULT NULL,
  `Weight` int DEFAULT NULL,
  `Eq_Name_FK` varchar(80) DEFAULT NULL,
  PRIMARY KEY (`Armor_Class`),
  KEY `Eq_Name_FK_idx` (`Eq_Name_FK`),
  CONSTRAINT `Eq_Name_FK2` FOREIGN KEY (`Eq_Name_FK`) REFERENCES `equipment` (`Eq_Name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `background` (
  `BG_Name` varchar(80) NOT NULL,
  `Story` varchar(45) DEFAULT NULL,
  `Job` varchar(45) DEFAULT NULL,
  `Characteristics` varchar(45) DEFAULT NULL,
  `C_Name_FK` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`BG_Name`),
  KEY `C_Name_FK_idx` (`C_Name_FK`),
  CONSTRAINT `C_Name_FK` FOREIGN KEY (`C_Name_FK`) REFERENCES `character` (`C_Name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `character` (
  `C_Name` varchar(80) NOT NULL,
  `PI_ID_FK` int DEFAULT NULL,
  PRIMARY KEY (`C_Name`),
  KEY `PI_ID_FK_idx` (`PI_ID_FK`),
  CONSTRAINT `PI_ID_FK` FOREIGN KEY (`PI_ID_FK`) REFERENCES `player` (`PI_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `class` (
  `Class_Name` varchar(80) NOT NULL,
  `Ability` varchar(45) DEFAULT NULL,
  `HP` int DEFAULT NULL,
  `Training` varchar(45) DEFAULT NULL,
  `Subclass` varchar(45) DEFAULT NULL,
  `Proficiency` int DEFAULT NULL,
  `C_name_FK2` varchar(45) DEFAULT NULL,
  `C_Name_FK` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`Class_Name`),
  KEY `C_Name_FK_idx` (`C_Name_FK`),
  KEY `C_name_FK2_idx` (`C_name_FK2`),
  CONSTRAINT `C_name_FK2` FOREIGN KEY (`C_name_FK2`) REFERENCES `character` (`C_Name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `equipment` (
  `Eq_Name` varchar(80) NOT NULL,
  `Price` int DEFAULT NULL,
  PRIMARY KEY (`Eq_Name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `misc` (
  `Usage` varchar(45) NOT NULL,

  `Eq_Name_FK` varchar(80) DEFAULT NULL,
  PRIMARY KEY (`Usage`),
  KEY `Eq_Name_FK_idx` (`Eq_Name_FK`),
  CONSTRAINT `Eq_Name_FK` FOREIGN KEY (`Eq_Name_FK`) REFERENCES `equipment` (`Eq_Name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `player` (
  `PI_ID` int NOT NULL,
  `PI_Name` varchar(80) DEFAULT NULL,
  PRIMARY KEY (`PI_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `race` (
  `Race_Name` varchar(80) NOT NULL,
  `Size` int DEFAULT NULL,
  `Language` varchar(45) DEFAULT NULL,
  `Abilities` varchar(45) DEFAULT NULL,
  `Traits` varchar(45) DEFAULT NULL,
  `Sub_races` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`Race_Name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `spells` (
  `Spell_Name` varchar(80) NOT NULL,
  `Type` varchar(45) DEFAULT NULL,
  `Level` int DEFAULT NULL,
  `Components` varchar(45) DEFAULT NULL,
  `Range` int DEFAULT NULL,
  `Duration` int DEFAULT NULL,
  `Casting_Time` int DEFAULT NULL,
  `Class_Name_FK` varchar(80) DEFAULT NULL,
  PRIMARY KEY (`Spell_Name`),
  KEY `Class_Name_FK_idx` (`Class_Name_FK`),
  CONSTRAINT `Class_Name_FK` FOREIGN KEY (`Class_Name_FK`) REFERENCES `class` (`Class_Name`)
